CREATE TABLE Users(
  netID VARCHAR(30),
  name  VARCHAR(100),
  password  VARCHAR(100),
  register_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  last_commit TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  blocked BOOLEAN DEFAULT FALSE
);

CREATE TABLE Attributes(
  netID VARCHAR(30),
  subject VARCHAR(10),
  courseNum INT,
  courseName  TEXT,
  category  VARCHAR(50),
  attributeName VARCHAR(100),
  score REAL,
  total REAL,
  dueDate VARCHAR(5),
  week INT
);

CREATE TABLE Weights(
  netID VARCHAR(30),
  subject VARCHAR(10),
  courseNum INT,
  courseName  TEXT,
  category  VARCHAR(50),
  weight  REAL,
  drops INT
);

CREATE TABLE Classes(
  netID VARCHAR(30),
  subject VARCHAR(10),
  courseNum INT,
  courseName  TEXT,
  weighted BOOLEAN
);
